1. PRISMA SCHEMA — tambah model baru ini:

model DeliveryOrder {
  id           String              @id @default(cuid())
  orderNumber  String              @unique
  date         DateTime
  driverId     String
  driver       User                @relation(fields: [driverId], references: [id])
  status       DeliveryOrderStatus @default(PENDING)
  recipients   DeliveryRecipient[]
  fuelReports  FuelReport[]
  activities   DeliveryActivity[]
  createdAt    DateTime            @default(now())
  updatedAt    DateTime            @updatedAt
}

model DeliveryRecipient {
  id              String           @id @default(cuid())
  deliveryOrderId String
  deliveryOrder   DeliveryOrder    @relation(fields: [deliveryOrderId], references: [id])
  name            String           // contoh: BU SURYANI - BSPS TLAHAB
  address         String
  phone           String?
  status          RecipientStatus  @default(PENDING)
  photos          DeliveryPhoto[]
  issues          DeliveryIssue[]
  signature       String?          // base64 PNG
  deliveredAt     DateTime?
  createdAt       DateTime         @default(now())
  updatedAt       DateTime         @updatedAt
}

model DeliveryPhoto {
  id          String            @id @default(cuid())
  recipientId String
  recipient   DeliveryRecipient @relation(fields: [recipientId], references: [id])
  type        PhotoType         // BARANG, SURAT_JALAN, PENERIMA, ISSUE, BBM
  url         String            // URL MinIO
  path        String            // path di MinIO: tanggal/ORDER-ID/NAMA-PENERIMA/barang.jpg
  uploadedAt  DateTime          @default(now())
}

model DeliveryIssue {
  id          String            @id @default(cuid())
  recipientId String
  recipient   DeliveryRecipient @relation(fields: [recipientId], references: [id])
  type        IssueType
  description String
  photoUrl    String?
  createdAt   DateTime          @default(now())
}

model FuelReport {
  id              String        @id @default(cuid())
  deliveryOrderId String?
  deliveryOrder   DeliveryOrder? @relation(fields: [deliveryOrderId], references: [id])
  driverId        String
  driver          User          @relation(fields: [driverId], references: [id])
  date            DateTime
  liters          Float
  pricePerLiter   Float
  total           Float
  spbu            String?
  receiptUrl      String?
  createdAt       DateTime      @default(now())
}

model DeliveryActivity {
  id              String        @id @default(cuid())
  deliveryOrderId String
  deliveryOrder   DeliveryOrder @relation(fields: [deliveryOrderId], references: [id])
  driverId        String
  action          String
  description     String
  createdAt       DateTime      @default(now())
}

enum DeliveryOrderStatus {
  PENDING
  IN_PROGRESS
  PARTIAL
  COMPLETED
}

enum RecipientStatus {
  PENDING
  ON_THE_WAY
  DELIVERED
  ISSUE_REPORTED
}

enum PhotoType {
  BARANG
  SURAT_JALAN
  PENERIMA
  ISSUE
  BBM
}

enum IssueType {
  PENERIMA_TIDAK_ADA
  ALAMAT_TIDAK_DITEMUKAN
  BARANG_RUSAK
  AKSES_JALAN_TERTUTUP
  LAINNYA
}

2. MINIO SERVICE
Buat src/modules/storage/minio.service.ts:
- Koneksi ke MinIO via env: MINIO_ENDPOINT, MINIO_PORT, MINIO_ACCESS_KEY, 
  MINIO_SECRET_KEY, MINIO_BUCKET
- Method uploadFile(file: Express.Multer.File, path: string): Promise<string>
  → Upload file ke path yang ditentukan
  → Return public URL file
- Method buildDeliveryPath(date: Date, orderNumber: string, recipientName: string, type: string): string
  → Return: 2026-06-12/ORDER-001/BU-SURYANI-BSPS-TLAHAB/barang.jpg
  → Nama penerima: spasi diganti strip, huruf kapital semua
- Method deleteFile(path: string): Promise<void>
- Bucket dibuat otomatis jika belum ada saat startup

4. ENDPOINTS LENGKAP:

GET /api/delivery/orders
- Query params: date (YYYY-MM-DD), driverId, status
- Guard: JWT Auth
- Return: list DeliveryOrder dengan jumlah recipient per status

GET /api/delivery/orders/:orderId
- Guard: JWT Auth  
- Return: DeliveryOrder lengkap dengan semua recipient + status foto per recipient

POST /api/delivery/orders
- Guard: JWT Auth + Role Admin/Manager
- Body: CreateDeliveryOrderDto
- Buat order baru + list recipient

PUT /api/delivery/orders/:orderId/status
- Guard: JWT Auth
- Update status order

PUT /api/delivery/orders/:orderId/recipients/:recipientId/status
- Guard: JWT Auth
- Body: { status: RecipientStatus }
- Update status recipient
- Jika status DELIVERED, set deliveredAt = now()
- Catat ke DeliveryActivity

POST /api/delivery/orders/:orderId/recipients/:recipientId/photos
- Guard: JWT Auth
- Content-Type: multipart/form-data
- Fields: barang (file), suratJalan (file), penerima (file), 
  signature (string base64, optional), timestamp (string)
- Validasi: ketiga foto wajib ada
- Upload masing-masing ke MinIO dengan path:
  tanggal/orderNumber/NAMA-PENERIMA/barang.jpg
  tanggal/orderNumber/NAMA-PENERIMA/surat-jalan.jpg
  tanggal/orderNumber/NAMA-PENERIMA/penerima.jpg
- Simpan URL ke DeliveryPhoto
- Catat ke DeliveryActivity: "Foto diunggah untuk [NAMA]"
- Return: { success: true, photos: [...urls] }

POST /api/delivery/orders/:orderId/recipients/:recipientId/issues
- Guard: JWT Auth
- Content-Type: multipart/form-data
- Fields: type, description, photo (file, optional)
- Upload foto ke MinIO jika ada
- Simpan ke DeliveryIssue
- Update status recipient jadi ISSUE_REPORTED
- Catat ke DeliveryActivity

POST /api/delivery/fuel-reports
- Guard: JWT Auth
- Content-Type: multipart/form-data
- Fields: date, liters, pricePerLiter, spbu, receipt (file optional)
- Upload struk ke MinIO jika ada
- Simpan ke FuelReport

GET /api/delivery/fuel-reports
- Guard: JWT Auth
- Query: month (YYYY-MM), driverId
- Return: list FuelReport bulan tersebut + total pengeluaran

GET /api/delivery/activity
- Guard: JWT Auth
- Query: date (YYYY-MM-DD), driverId
- Return: list DeliveryActivity urut dari terbaru
- Format deskripsi: "09:15 - Foto diunggah untuk Bu Suryani"

GET /api/health
- Public endpoint (no auth)
- Return: { status: 'ok', timestamp: ISO string, version: '1.0.0' }
